cmake_minimum_required(VERSION 2.8)

project(agast)

find_package(redwood_catkin REQUIRED)
catkin_simple()

add_definitions(-Wextra -Wall -Werror -pedantic)

file(GLOB AGAST_SOURCE_FILES  "${PROJECT_SOURCE_DIR}/src/*.cc")
file(GLOB AGAST_HEADER_FILES  "${PROJECT_SOURCE_DIR}/include/agast/*.h")

cs_add_library(${PROJECT_NAME} ${AGAST_SOURCE_FILES})
     
cs_install()

cs_export()
redwood_generate_ndk_import_module(${PROJECT_NAME})
